주소 검색 라이브러리 react-daum-postcode

react-daum-postcode는 React 환경에서 Daum 우편번호 검색 서비스를 간편하게 이용할 수 있는 라이브러리 패키지입니다.

Install

npm install react-daum-postcode
yarn add react-daum-postcode

Embed

DaumPostcodeEmbed 컴포넌트를 사용하여, 우편번호 검색 서비스를 임베드 방식으로 사용할 수 있습니다.

// Postcode.jsx
import React from "react";
import DaumPostcodeEmbed from "react-daum-postcode";

const Postcode = ({ address, setAddress }) => {
  const handleComplete = (data) => {
    let fullAddress = data.address;
    let extraAddress = "";

    // 도로명 주소와 함께 법정동/법정리/건물명이 있는 경우 전체 주소로 합쳐주는 과정
    if (data.addressType === "R") {
      if (data.bname !== "") {
        extraAddress += data.bname;
      }
      if (data.buildingName !== "") {
        extraAddress +=
          extraAddress !== "" ? `, ${data.buildingName}` : data.buildingName;
      }
      fullAddress += extraAddress !== "" ? ` (${extraAddress})` : "";
    }

    setAddress({
      ...address,
      fullAddress: fullAddress,
      postcode: data.zonecode,
    });
  };

  return <DaumPostcodeEmbed onComplete={handleComplete} />;
};

// AddressForm.jsx
import { useState } from "react";
import Postcode from "./Postcode";

const AddressForm = () => {
  const [modalOpen, setModalOpen] = useState(false);
  const [address, setAddress] = useState({
    fullAddress: undefined,
    postcode: undefined,
    detailAddress: undefined,
  });
  return (
    <form>
      <input
        type="number"
        placeholder="우편번호"
        value={address.postcode}
        readOnly
      />
      <input
        type="text"
        placeholder="주소"
        value={address.fullAddress}
        readOnly
      />
      <input
        type="text"
        placeholder="상세주소"
        value={address.detailAddress}
        onChange={(e) =>
          setAddress({ ...address, detailAddress: e.target.value })
        }
      />
      <button onClick={() => setModalOpen(!modalOpen)}>주소찾기</button>
      {modalOpen && <Postcode address={address} setAddress={setAddress} />}
    </form>
  );
};

postcode의 자세한 생성자 속성들은 Daum 우편번호 서비스 가이드를 참고해주세요!
